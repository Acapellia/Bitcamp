void BookReader::editVocaScore(char tword[30], int vScore) {
	FILE* in = fopen("Voca.txt", "r+");
	int n = 0,idx=0;
	char word[30], mean[30];
	int score=0;
	while (!feof(in)) {
		fscanf(in, "%s", word); fscanf(in, "%s", mean); fscanf(in, "%d", &score);
		if (!strcmp(word, tword)) {
			n += (strlen(word) + strlen(mean) + 2);
			voca[idx].setScore((vScore==1||vScore==-1)?score:vScore);
			break;
		}
		n += (strlen(word) + strlen(mean) + 2);
		n += (strlen("\n") + 1);
		idx++;
		//cout << word << " " << mean << endl;
	}
	fseek(in, n, SEEK_SET);

	if (vScore == 1 || vScore == -1) {
		fprintf(in, " %d", score + vScore);
	}
	else {
		score >= 10 ? fprintf(in, "%d ", vScore) : fprintf(in, "%d", vScore);
	}

	fclose(in);
}
